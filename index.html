🔧 신규 DB SQL
-- To-do 테이블
create table if not exists todos (
  id uuid default gen_random_uuid() primary key,
  employee_id uuid references employees(id),
  content text not null,
  is_done boolean default false,
  done_at timestamptz,
  source text default 'personal',
  due_date date,
  note text,
  checklist jsonb,
  assignee_ids text,
  assignee_names text,
  parent_todo_id uuid,
  created_at timestamptz default now()
);
-- 기존 todos 테이블에 컬럼 추가 (이미 생성된 경우)
alter table todos add column if not exists due_date date;
alter table todos add column if not exists note text;
alter table todos add column if not exists checklist jsonb;
alter table todos add column if not exists assignee_ids text;
alter table todos add column if not exists assignee_names text;
alter table todos add column if not exists parent_todo_id uuid;
-- 공지 테이블
create table if not exists notices (
  id uuid default gen_random_uuid() primary key,
  title text not null,
  body text not null,
  created_by uuid references employees(id),
  created_at timestamptz default now()
);
-- 근태 테이블
create table if not exists attends (
  id uuid default gen_random_uuid() primary key,
  employee_id uuid references employees(id),
  start_date date not null,
  end_date date not null,
  attend_type text not null,
  note text,
  exclude_hours numeric(5,1) default 0,
  created_at timestamptz default now()
);
-- work_logs 컬럼 추가
alter table work_logs
  add column if not exists start_time text,
  add column if not exists end_time text;
-- todos 순서 컬럼 추가
alter table todos add column if not exists sort_order integer default 0;
alter table todos add column if not exists title text;
alter table todos add column if not exists status text default 'waiting';
-- 템플릿 테이블 (localStorage 대신 DB에 저장)
create table if not exists user_templates (
  id uuid default gen_random_uuid() primary key,
  employee_id uuid references employees(id),
  category_id text,
  cat_name text,
  content text not null,
  created_at timestamptz default now()
);
alter table user_templates enable row level security;
create policy "allow_all" on user_templates for all using (true);
-- RLS
alter table todos enable row level security;
alter table notices enable row level security;
alter table attends enable row level security;
create policy "allow_all" on todos for all using (true);
create policy "allow_all" on notices for all using (true);
create policy "allow_all" on attends for all using (true);
